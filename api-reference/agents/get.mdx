---
title: 'Get Agent'
api: 'GET /api/v1/agents/{agent_id}'
description: 'Retrieve details of a specific agent'
---

## Path Parameters

<ParamField path="agent_id" type="string" required>
  The unique identifier of the agent to retrieve
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the agent
</ResponseField>

<ResponseField name="name" type="string">
  The name of the agent
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the agent was created
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the agent: `initializing`, `ready`, `error`
</ResponseField>

<ResponseField name="model" type="object">
  <Expandable title="properties">
    <ResponseField name="provider" type="string">
      The AI model provider
    </ResponseField>
    <ResponseField name="version" type="string">
      The specific model version in use
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="configuration" type="object">
  <Expandable title="properties">
    <ResponseField name="temperature" type="number">
      Model temperature setting
    </ResponseField>
    <ResponseField name="max_tokens" type="integer">
      Maximum tokens per response
    </ResponseField>
    <ResponseField name="timeout_seconds" type="integer">
      Request timeout in seconds
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="capabilities" type="array">
  List of enabled capabilities
</ResponseField>

## Example Request

```bash
curl https://api.tryiris.dev/api/v1/agents/agt_7f8d9e2a1b3c \
  -H "Authorization: Bearer ${API_KEY}"
```

## Example Response

```json
{
  "id": "agt_7f8d9e2a1b3c",
  "name": "e2e-test-agent",
  "created_at": "2024-01-20T08:30:00Z",
  "status": "ready",
  "model": {
    "provider": "vertex-ai",
    "version": "text-bison-32k"
  },
  "configuration": {
    "temperature": 0.7,
    "max_tokens": 1024,
    "timeout_seconds": 300
  },
  "capabilities": [
    "ui-interaction",
    "test-generation",
    "self-healing"
  ]
}
```

## Error Responses

  ```json
  {
    "error": "not_found",
    "message": "Agent not found"
  }
  ```
</ResponseField>

<ResponseField name="401 Unauthorized" type="object">

  ```json
  {
    "error": "unauthorized",
    "message": "Invalid API key provided"
  }
  ```
</ResponseField>

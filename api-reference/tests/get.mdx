---
title: 'Get Test'
api: 'GET /api/v1/tests/{test_id}'
description: 'Retrieve details of a specific test'
---

## Path Parameters

<ParamField path="test_id" type="string" required>
  The unique identifier of the test to retrieve
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the test
</ResponseField>

<ResponseField name="project_id" type="string">
  The project identifier
</ResponseField>

<ResponseField name="name" type="string">
  The name of the test
</ResponseField>

<ResponseField name="description" type="string">
  The test description
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the test was created
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the test was last updated
</ResponseField>

<ResponseField name="steps" type="array">
  <Expandable title="properties">
    <ResponseField name="type" type="string">
      Type of step
    </ResponseField>
    <ResponseField name="description" type="string">
      Step description
    </ResponseField>
    <ResponseField name="selector" type="string">
      Element selector
    </ResponseField>
    <ResponseField name="value" type="string">
      Step value
    </ResponseField>
    <ResponseField name="timeout" type="integer">
      Step timeout
    </ResponseField>
    <ResponseField name="screenshot" type="boolean">
      Screenshot setting
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tags" type="array">
  List of test tags
</ResponseField>

<ResponseField name="configuration" type="object">
  <Expandable title="properties">
    <ResponseField name="retry_on_failure" type="boolean">
      Retry setting
    </ResponseField>
    <ResponseField name="max_retries" type="integer">
      Maximum retries
    </ResponseField>
    <ResponseField name="retry_interval" type="integer">
      Retry interval
    </ResponseField>
    <ResponseField name="timeout" type="integer">
      Global timeout
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="stats" type="object">
  <Expandable title="properties">
    <ResponseField name="total_runs" type="integer">
      Total number of test runs
    </ResponseField>
    <ResponseField name="success_rate" type="number">
      Percentage of successful runs
    </ResponseField>
    <ResponseField name="avg_duration" type="number">
      Average run duration in seconds
    </ResponseField>
    <ResponseField name="last_run" type="object">
      Details of the most recent run
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

```bash
curl https://api.tryiris.dev/api/v1/tests/tst_1a2b3c4d5e \
  -H "Authorization: Bearer ${API_KEY}"
```

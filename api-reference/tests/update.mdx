---
title: 'Update Test'
api: 'PUT /api/v1/tests/{test_id}'
description: 'Update an existing test configuration'
---

## Path Parameters

<ParamField path="test_id" type="string" required>
  The unique identifier of the test to update
</ParamField>

## Request Body

<ParamField body="name" type="string">
  New name for the test
</ParamField>

<ParamField body="description" type="string">
  Updated test description
</ParamField>

<ParamField body="steps" type="array">
  <Expandable title="properties">
    <ParamField body="type" type="string">
      Type of step: `navigate`, `click`, `type`, `assert`, `wait`, `custom`
    </ParamField>
    <ParamField body="description" type="string">
      Human-readable description of what the step does
    </ParamField>
    <ParamField body="selector" type="string">
      CSS or XPath selector for element interaction steps
    </ParamField>
    <ParamField body="value" type="string">
      Value to type or assert against
    </ParamField>
    <ParamField body="timeout" type="integer">
      Step-specific timeout in seconds
    </ParamField>
    <ParamField body="screenshot" type="boolean">
      Whether to capture a screenshot after this step
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="tags" type="array">
  Updated list of tags
</ParamField>

<ParamField body="configuration" type="object">
  <Expandable title="properties">
    <ParamField body="retry_on_failure" type="boolean">
      Whether to retry failed steps
    </ParamField>
    <ParamField body="max_retries" type="integer">
      Maximum number of retries per step
    </ParamField>
    <ParamField body="retry_interval" type="integer">
      Seconds to wait between retries
    </ParamField>
    <ParamField body="timeout" type="integer">
      Global timeout for the test in seconds
    </ParamField>
  </Expandable>
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the test
</ResponseField>

<ResponseField name="name" type="string">
  Updated name of the test
</ResponseField>

<ResponseField name="description" type="string">
  Updated test description
</ResponseField>

<ResponseField name="updated_at" type="string">
  ISO 8601 timestamp of when the test was updated
</ResponseField>

<ResponseField name="steps" type="array">
  Updated array of test steps
</ResponseField>

<ResponseField name="tags" type="array">
  Updated list of test tags
</ResponseField>

<ResponseField name="configuration" type="object">
  Updated test configuration
</ResponseField>

## Example Request

```bash
curl -X PUT https://api.tryiris.dev/api/v1/tests/tst_1a2b3c4d5e \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "Enhanced Product Purchase Flow",
    "description": "Updated end-to-end product purchase test with additional validations",
    "steps": [
      {
        "type": "navigate",
        "description": "Navigate to product page",
        "value": "/products/demo-product"
      },
      {
        "type": "assert",
        "description": "Verify product price",
        "selector": ".product-price",
        "value": "$99.99"
      },
      {
        "type": "click",
        "description": "Click Add to Cart button",
        "selector": "#add-to-cart-button"
      },
      {
        "type": "wait",
        "description": "Wait for cart update",
        "selector": ".cart-count[data-count='1']",
        "timeout": 5
      },
      {
        "type": "assert",
        "description": "Verify product added to cart",
        "selector": ".cart-item",
        "value": "Demo Product"
      }
    ],
    "tags": ["e-commerce", "critical-path", "purchase-flow"],
    "configuration": {
      "retry_on_failure": true,
      "max_retries": 5,
      "retry_interval": 3,
      "timeout": 400
    }
  }'
```

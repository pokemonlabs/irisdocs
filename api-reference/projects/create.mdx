---
title: 'Create Project'
api: 'POST /api/v1/projects'
description: 'Create a new test project'
---

## Request Body

<ParamField body="name" type="string" required>
  The name of the project. Must be unique within your organization.
</ParamField>

<ParamField body="description" type="string">
  A detailed description of the project and its testing objectives.
</ParamField>

<ParamField body="settings" type="object">
  <Expandable title="properties">
    <ParamField body="base_url" type="string" required>
      The base URL of the application to be tested
    </ParamField>
    <ParamField body="default_timeout" type="integer">
      Default timeout in seconds for test executions. Default: 30
    </ParamField>
    <ParamField body="environment" type="string">
      Target environment (e.g., 'development', 'staging', 'production')
    </ParamField>
    <ParamField body="viewport" type="object">
      Browser viewport settings
      <Expandable title="properties">
        <ParamField body="width" type="integer">
          Viewport width in pixels. Default: 1920
        </ParamField>
        <ParamField body="height" type="integer">
          Viewport height in pixels. Default: 1080
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="tags" type="array">
  List of tags for organizing projects
</ParamField>

## Response

<ResponseField name="id" type="string">
  The unique identifier of the created project
</ResponseField>

<ResponseField name="name" type="string">
  The name of the project
</ResponseField>

<ResponseField name="description" type="string">
  The project description
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the project was created
</ResponseField>

<ResponseField name="settings" type="object">
  Project configuration settings
</ResponseField>

<ResponseField name="tags" type="array">
  List of project tags
</ResponseField>

## Example Request

```bash
curl -X POST https://api.tryiris.dev/api/v1/projects \
  -H "Authorization: Bearer ${API_KEY}" \
  -H "Content-Type: application/json" \
  -d '{
    "name": "e-commerce-testing",
    "description": "End-to-end testing for our e-commerce platform",
    "settings": {
      "base_url": "https://shop.example.com",
      "default_timeout": 45,
      "environment": "staging",
      "viewport": {
        "width": 1920,
        "height": 1080
      }
    },
    "tags": ["e-commerce", "web", "staging"]
  }'
```
